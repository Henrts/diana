### Building a new Button

Extend the BaseButton by using the `extendStyles` property:

```javascript
const DerivedButton = BaseButton.extendStyles((theme) => ({
  button: {
    /* Derived Button default styling */
  },
  disabled: {
    /* Derived Button disabled styling */
  }
});
```

### Advanced Button styling

Advanced Button customization is also available using props like `renderIconLeft`; In this example, we use this prop to build a Button with a Icon left of the Button text.

```javascript
const DerivedButton: React.FC = ({ children, ...props }) => {
  const AlteredBaseButton = BaseButton.extendStyles((theme: Theme) => ({
    button: {
      backgroundColor: "steelblue",
      color: "white",
      padding: "24px",
      borderRadius: "10px"
    }
  }));
  const renderLeftIcon = () => <Icon stroke="white" size={10} name="add" />;
  return (
    <AlteredBaseButton renderLeftIcon={renderLeftIcon} {...props}>
      {children}
    </AlteredBaseButton>
  );
};
```
